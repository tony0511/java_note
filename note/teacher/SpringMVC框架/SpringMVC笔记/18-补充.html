url
/role/deleteMore?ids=1&ids=2&ids=3

controller
    @ResponseBody
    @RequestMapping("/role/deleteMore")
    public ResultBean getAll(@RequestParam(value="ids") List<Integer> ids){
        //return roleService.getRoleList();
        return null;
    }
或
    @ResponseBody
    @RequestMapping("/role/deleteMore")
    public ResultBean getAll(@RequestParam(value="ids") String[] ids){
        //return roleService.getRoleList();
        return null;
    }
