2、定义js
注意：请求数据中发送的是json 字符串
所以传输之前要将javascript对象转换成json字符串
    $(function(){
        $('#saveBtn').click(function(){
            var row = {
                    username: $('#username').val(),
                    password: $('#password').val(),
                    age: $('#age').val(),
            }
            $.ajax({
                url: 'user-json',
                type: 'post' ,
                contentType: 'application/json;charset=utf-8',
                data: JSON.stringify(row)
            });
        });
    });
