配置文件：
注意：
1、一定要有鉴别器，测试才能通过
2、查询结果中一定要包含鉴别器列
    <resultMap type="Vehicle" id="vehicleResultMap">
        <id column="id" property="id"/>
        <discriminator javaType="int" column="vehicle_type">
            <case value="1" resultType="Car">
                <result property="doorCount" column="door_count" />
            </case>
            <case value="2" resultType="Suv">
                <result property="allWheelDrive" column="all_wheel_drive" />
            </case>
        </discriminator>
    </resultMap>

    <select id="selectVehicleById" parameterType="int" 
    resultMap="vehicleResultMap">
        SELECT * FROM Vehicle WHERE id = #{id}
    </select>
    <select id="selectVehicleList" resultMap="vehicleResultMap">
        SELECT * FROM Vehicle
    </select>
接口：
public interface VehicleMapper {
    public Vehicle selectVehicleById(Integer id);
    public List<Vehicle> selectVehicleList();
}
