Mybaits官方提供逆向工程，根据数据库表结构生成mapper.xml和mapper.java及pojo类。
生成的mapper具备了单表的增、删、改、查的功能。
使用java程序+xml配置执行逆向工程。

下载地址：

https://github.com/mybatis/generator/releases/tag/mybatis-generator-1.3.2


1、引入jar包

2、编写xml配置文件

3、编写java代码

4、运行主方法生成相关文件

5、将文件移植到项目中



注意：

Mapper.xml文件已经存在时，重新生成后xml文件的内容不被覆盖而是进行内容追加，结果导致mybatis解析失败。

解决方法：删除原来已经生成的mapper xml文件再进行生成。

Mybatis自动生成的po及mapper.java文件不是内容而是直接覆盖没有此问题。



public class AuthorMapperTest {
    SqlSession session = null;
    @Before
    public void init(){
        //System.out.println("init");
        session = MyBatisUtil.getSqlSession();
    }
    @After
    public void close(){
        //System.out.println("close");
        session.close();
    }
    //插入用户信息
    @Test
    public void testInsert() {
        AuthorMapper authorDao = session.getMapper(AuthorMapper.class);
        Author author = new Author();
        author.setEmail("123");
        author.setFavouriteSection("ff");
        author.setPassword("ppp");
        author.setUsername("uuu");
        authorDao.insert(author);
        
        session.commit();
    }
    
    //插入用户信息(动态sql)
    @Test
    public void testInsertSelective() {
        AuthorMapper authorDao = session.getMapper(AuthorMapper.class);
        Author author = new Author();
        author.setPassword("ppp");
        author.setUsername("uuu");
        authorDao.insertSelective(author);
        
        session.commit();
    }
    //根据主键删除
    @Test
    public void testDeleteByPrimaryKey() {
        
        AuthorMapper authorDao = session.getMapper(AuthorMapper.class);
        int count = authorDao.deleteByPrimaryKey(9);
        
        session.commit();
        System.out.println("删除了" + count + "条");
    }
    
    //根据主键删除（根据查询条件删除）
    @Test
    public void testDeleteByExample() {
        
        AuthorMapper authorDao = session.getMapper(AuthorMapper.class);
        
        AuthorExample authorExample = new AuthorExample();
        AuthorExample.Criteria criteria = authorExample.createCriteria();
        criteria.andUsernameLike("%u%");
        
        int count = authorDao.deleteByExample(authorExample);
        
        session.commit();
        System.out.println("删除了" + count + "条");
    }
    //根据主键更新，先查询再更新
    @Test
    public void testUpdateByPrimaryKey() {
            
        AuthorMapper authorDao = session.getMapper(AuthorMapper.class);
        
        Author author = authorDao.selectByPrimaryKey(6);
        author.setPassword("p1");
        author.setUsername("u1");
        
        int count = authorDao.updateByPrimaryKey(author);
        session.commit();
        
        System.out.println("更新了" + count + "条记录");
    }
    //根据主键更新，只更新设置的字段
    @Test
    public void testUpdateByPrimaryKeySelective() {
        AuthorMapper authorDao = session.getMapper(AuthorMapper.class);
        Author author = new Author();
        author.setPassword("p1");
        author.setUsername("u1");
        author.setId(4);
        
        int count = authorDao.updateByPrimaryKeySelective(author);
        session.commit();
        
        System.out.println("更新了" + count + "条记录");
    }
    
    //根据查询模版更新
    @Test
    public void testUpdateByExampleSelective() {
        AuthorMapper authorDao = session.getMapper(AuthorMapper.class);
        Author author = new Author();
        author.setPassword("p100");
        author.setUsername("u100");
        
        AuthorExample authorExample = new AuthorExample();
        AuthorExample.Criteria criteria = authorExample.createCriteria();
        criteria.andUsernameLike("%u%");
        
        int count = authorDao.updateByExampleSelective(author, authorExample);
        session.commit();
        
        System.out.println("更新了" + count + "条记录");
    }
    //根据主键查询用户信息
    @Test
    public void testSelectByPrimaryKey() {
        
        AuthorMapper authorDao = session.getMapper(AuthorMapper.class);
        Author author = authorDao.selectByPrimaryKey(2);
        System.out.println(author.getUsername());
    }
    //自定义查询条件查询
    @Test
    public void testSelectByExample() {
        AuthorMapper authorDao = session.getMapper(AuthorMapper.class);
        AuthorExample authorExample = new AuthorExample();
        AuthorExample.Criteria criteria = authorExample.createCriteria();
        criteria.andEmailLike("%3%");
        criteria.andUsernameEqualTo("u100");
        
        List<Author> authorList = authorDao.selectByExample(authorExample);
        
        for (Author author : authorList) {
            System.out.println(author.getEmail());
        }
        
    }
}
